Supabase 테이블 생성 SQL

아직 테이블이 없다면 Supabase → SQL Editor에서 아래 SQL을 실행하세요.
create table locations ( id uuid default gen_random_uuid() primary key, name text not null, is_active boolean default true, created_at timestamptz default now() ); create table approvers ( id uuid default gen_random_uuid() primary key, name text not null, pin text not null, created_at timestamptz default now() ); create table move_sessions ( id uuid default gen_random_uuid() primary key, session_date date not null, from_location text not null, to_location text not null, status text default 'pending', created_by text not null, approved_by text, approved_at timestamptz, is_correction boolean default false, correction_reason text, created_at timestamptz default now() ); create table session_items ( id uuid default gen_random_uuid() primary key, session_id uuid references move_sessions(id), barcode text not null, product_name text, category text, price text, current_location text, scanned_by text not null, scanned_at timestamptz default now(), is_error boolean default false, error_msg text ); alter publication supabase_realtime add table session_items;
